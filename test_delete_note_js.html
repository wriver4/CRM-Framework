Test Delete Note JavaScript

This page tests the JavaScript delete functionality without actually deleting notes.

Instructions:
  1. Open browser developer tools (F12)
  2. Go to the Console tab
  3. Run the test functions below

Test Functions

1. Test AJAX Request Format

// Test the AJAX request format
function testAjaxRequest() {
    const formData = new FormData();
    formData.append('note_id', 999999);
    formData.append('lead_id', 1);
    
    console.log('Testing AJAX request...');
    
    fetch('/admin/leads/delete_note.php', {
        method: 'POST',
        body: formData
    })
    .then(response => {
        console.log('Response status:', response.status);
        console.log('Response headers:', response.headers);
        return response.text();
    })
    .then(text => {
        console.log('Raw response:', text);
        try {
            const data = JSON.parse(text);
            console.log('Parsed JSON:', data);
        } catch (e) {
            console.error('JSON parse error:', e);
        }
    })
    .catch(error => {
        console.error('Fetch error:', error);
    });
}

// Run the test
testAjaxRequest();

2. Test DOM Manipulation

// Test DOM manipulation without actual deletion
function testDOMManipulation() {
    // Create a fake timeline item
    const testItem = document.createElement('div');
    testItem.className = 'timeline-item mb-4';
    testItem.innerHTML = `
        <div class="timeline-content">
            <div class="note-text">Test note for DOM manipulation</div>
            <button class="btn btn-outline-danger btn-sm delete-note-btn" 
                    data-note-id="test" data-lead-id="1">
                Delete
            </button>
        </div>
    `;
    
    // Add to page
    document.body.appendChild(testItem);
    
    // Test finding the element
    const button = testItem.querySelector('.delete-note-btn');
    const noteElement = button.closest('.timeline-item');
    
    console.log('Button found:', button);
    console.log('Note element found:', noteElement);
    
    // Test animation
    noteElement.style.transition = 'opacity 0.3s ease-out';
    noteElement.style.opacity = '0';
    
    setTimeout(() => {
        noteElement.remove();
        console.log('Test element removed successfully');
    }, 300);
}

// Run the test
testDOMManipulation();
